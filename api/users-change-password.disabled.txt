const { findOne, updateOne } = require('./_lib/mongodb');
const {
  hashPassword,
  validatePassword,
  validateUsername,
  verifyPassword,
  verifySession,
} = require('./_lib/security');

module.exports = async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  const { username, authToken, currentPassword, newPassword } = req.body || {};

  let normalizedUsername;
  let validatedNewPassword;
  try {
    normalizedUsername = validateUsername(username);
    validatedNewPassword = validatePassword(newPassword);
    if (!currentPassword || `${currentPassword}`.length > 128) {
      throw new Error('Current password is required.');
    }
  } catch (error) {
    return res.status(400).json({ error: error.message });
  }

  if (!authToken) {
    return res.status(400).json({ error: 'Username and auth token are required.' });
  }

  try {
    const user = await findOne({ username: normalizedUsername });
    if (!user) {
      return res.status(404).json({ error: 'User not found.' });
    }

    if (!verifySession(user, authToken)) {
      return res.status(401).json({ error: 'Unauthorized session.' });
    }

    const storedPassword = user.passwordHash || user.password;
    if (!verifyPassword(`${currentPassword}`, storedPassword)) {
      return res.status(401).json({ error: 'Current password is wrong.' });
    }

    await updateOne(
      { username: normalizedUsername },
      {
        $set: { passwordHash: hashPassword(validatedNewPassword) },
        $unset: { password: '' },
      },
      { upsert: false }
    );

    return res.status(200).json({ ok: true });
  } catch (error) {
    return res.status(500).json({ error: error.message || 'Unable to change password.' });
  }
};
